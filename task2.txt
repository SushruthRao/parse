FileTransactionService.java — concatenate logs in getRxInfo():
javaprivate RxInfo getRxInfo(RawJsonDataDto data, MembershipInfo member) {
    String combinedLog = String.join(" | ", data.log1(), data.log2(), data.log3(), data.log4(), data.log5());

    RxInfo rxInfo = new RxInfo();
    rxInfo.setRx(data.rx());
    rxInfo.setDrugName(data.drugName());
    rxInfo.setDescription(data.description());
    rxInfo.setLog(combinedLog);
    rxInfo.setMember(member);
    return rxInfo;
}
