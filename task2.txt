@Transactional(Transactional.TxType.REQUIRED)   // can be removed, but harmless
public void saveRecordsInBatch(List<IndexedBatchingRecord> indexedRecords) {

    int batchSize = properties.getBatch().getBatchSize();
    log.info("[FileBatchingService] Pass 1: Trying to insert {} records in parallel (batch size ignored here)",
             indexedRecords.size());

    try {
        // Pre‑validation – keep this
        for (IndexedBatchingRecord indexed : indexedRecords) {
            if (!indexed.isValid()) {
                throw new InvalidRecordException("Invalid record found in batch at line " + indexed.lineNumber());
            }
        }

        // Parallel execution – each record in its own transaction
        List<CompletableFuture<Void>> futures = indexedRecords.stream()
            .map(record -> CompletableFuture.runAsync(() -> {
                fileTransactionService.saveSingleRecord(record.data().memberId(), record);
            }, batchExecutor))
            .collect(Collectors.toList());

        // Wait for all; any exception will cause a CompletionException to be thrown here
        CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();

        log.info("[FileBatchingService] Managed to save all records in parallel successfully");

    } catch (InvalidRecordException invalidRecordException) {
        log.warn("[FileBatchingService] Skipping batch because of invalid record: {}",
                 invalidRecordException.getMessage());
        processRecordsIndividually(indexedRecords, batchSize);

    } catch (Exception exception) {
        // This also catches CompletionException from parallel failures
        log.error("[FileBatchingService] Batch failed, falling back to individual processing", exception);
        processRecordsIndividually(indexedRecords, batchSize);
    }
}
