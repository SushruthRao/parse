package com.project.rxparser.service.impl;

import com.project.rxparser.dto.RawJsonDataDto;
import com.project.rxparser.model.MembershipInfo;
import com.project.rxparser.model.RxInfo;
import com.project.rxparser.repository.MembershipInfoRepository;
import jakarta.transaction.Transactional;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;

@Service
@Slf4j
public class RxBatchSaverService {

    private final MembershipInfoRepository membershipRepository;

    public RxBatchSaverService(MembershipInfoRepository membershipRepository) {
        this.membershipRepository = membershipRepository;
    }

    /**
     * Saves a single member's records in its own REQUIRES_NEW transaction.
     * If this fails, only this member's batch is rolled back.
     */
    @Transactional(Transactional.TxType.REQUIRES_NEW)
    public void saveMemberRecords(String memberId, List<RawJsonDataDto> records) {
        MembershipInfo member = getExistingMemberOrCreateMember(records.getFirst());

        Set<String> existingRx = member.getRxInfo().stream()
                .map(RxInfo::getRx)
                .collect(Collectors.toSet());

        List<RxInfo> newRxList = records.stream()
                .filter(r -> !existingRx.contains(r.rx()))
                .map(r -> buildRxInfo(r, member))
                .toList();

        member.getRxInfo().addAll(newRxList);
        membershipRepository.save(member);
    }

    private MembershipInfo getExistingMemberOrCreateMember(RawJsonDataDto record) {
        return membershipRepository.findById(Long.parseLong(record.memberId()))
                .orElseGet(() -> {
                    MembershipInfo m = new MembershipInfo();
                    m.setMemberId(Long.parseLong(record.memberId()));
                    m.setFirstName(record.firstname());
                    m.setLastName(record.lastname());
                    m.setDob(record.dob());
                    return m;
                });
    }

    private RxInfo buildRxInfo(RawJsonDataDto record, MembershipInfo member) {
        RxInfo rxInfo = new RxInfo();
        rxInfo.setRx(record.rx());
        rxInfo.setDrugName(record.drugName());
        rxInfo.setDescription(record.description());
        rxInfo.setMember(member);
        return rxInfo;
    }
}

private void saveToDatabase(List<RawJsonDataDto> rawDataList) {

    // Group all records by memberId first (preserving order)
    Map<String, List<RawJsonDataDto>> groupedByMemberId = rawDataList.stream()
            .collect(Collectors.groupingBy(RawJsonDataDto::memberId,
                    LinkedHashMap::new,
                    Collectors.toList()));

    List<Map.Entry<String, List<RawJsonDataDto>>> entries = new ArrayList<>(groupedByMemberId.entrySet());

    // Partition member entries into batches of batchSize
    for (int i = 0; i < entries.size(); i += batchSize) {
        List<Map.Entry<String, List<RawJsonDataDto>>> batch =
                entries.subList(i, Math.min(i + batchSize, entries.size()));

        log.info("[RxServiceImpl.saveToDatabase] Processing batch {}-{} of {}",
                i + 1, i + batch.size(), entries.size());

        for (Map.Entry<String, List<RawJsonDataDto>> entry : batch) {
            try {
                rxBatchSaverService.saveMemberRecords(entry.getKey(), entry.getValue());
            } catch (Exception e) {
                log.error("[RxServiceImpl.saveToDatabase] Failed to save memberId={}: {}",
                        entry.getKey(), e.getMessage());
                // Continue — other records in this batch are unaffected
            }
        }
    }
}

###### Why this works



| Concern | Solution |
| --- | --- |
| Batching | `entries.subList(i, i + batchSize)` partitions member groups into chunks; `batchSize` is `@Value`-injected from properties |
| Fault isolation | `REQUIRES_NEW` on `saveMemberRecords` suspends the outer transaction and opens a fresh one per member, so a failure only rolls back that one member |
| Self-invocation trap | `RxBatchSaverService` is a separate Spring bean — Spring's proxy intercepts the `REQUIRES_NEW` correctly, unlike calling `this.method()` |
| Configurability | Change `rx.batch.size` in `application.properties` with no code changes |



> **Note:** "batch of 100" here means 100 **members** per loop iteration, each committed independently. If you want 100 **rows** (across members), you'd need to flatten first and re-group — but per-member grouping is almost always the right semantic for this kind of data.

###### 3. `RxBatchSaverService.java` — true JDBC batch insert with per-record fault isolation

 java

```java
package com.project.rxparser.service.impl;

import com.project.rxparser.dto.RawJsonDataDto;
import com.project.rxparser.model.MembershipInfo;
import com.project.rxparser.model.RxInfo;
import com.project.rxparser.repository.MembershipInfoRepository;
import jakarta.persistence.EntityManager;
import jakarta.transaction.Transactional;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

import java.util.*;
import java.util.stream.Collectors;

@Service
@Slf4j
public class RxBatchSaverService {

    private final MembershipInfoRepository membershipRepository;
    private final EntityManager entityManager;

    @Value("${rx.batch.size:100}")
    private int batchSize;

    public RxBatchSaverService(MembershipInfoRepository membershipRepository,
                               EntityManager entityManager) {
        this.membershipRepository = membershipRepository;
        this.entityManager = entityManager;
    }

    /**
     * Inserts all valid records into DB using JDBC batch inserts.
     * Records are partitioned into configurable batch sizes.
     * If one record fails, it is captured and the rest continue.
     *
     * @param rawDataList all valid parsed records
     * @return list of memberIds that failed to save
     */
    public List<String> batchInsert(List<RawJsonDataDto> rawDataList) {

        // Group by memberId so we build MembershipInfo + RxInfo together
        Map<String, List<RawJsonDataDto>> groupedByMemberId = rawDataList.stream()
                .collect(Collectors.groupingBy(RawJsonDataDto::memberId,
                        LinkedHashMap::new,
                        Collectors.toList()));

        List<Map.Entry<String, List<RawJsonDataDto>>> entries =
                new ArrayList<>(groupedByMemberId.entrySet());

        List<String> failedMemberIds = new ArrayList<>();

        // Partition into batches of batchSize
        for (int i = 0; i < entries.size(); i += batchSize) {
            List<Map.Entry<String, List<RawJsonDataDto>>> batch =
                    entries.subList(i, Math.min(i + batchSize, entries.size()));

            log.info("[RxBatchSaverService.batchInsert] Inserting batch {}-{} of {} members",
                    i + 1, i + batch.size(), entries.size());

            List<String> batchFailures = saveBatch(batch);
            failedMemberIds.addAll(batchFailures);
        }

        return failedMemberIds;
    }

    /**
     * Saves one batch. Each member record is attempted individually —
     * a failure on one does not affect others in the batch.
     */
    @Transactional(Transactional.TxType.REQUIRES_NEW)
    public List<String> saveBatch(List<Map.Entry<String, List<RawJsonDataDto>>> batch) {

        List<String> failedInBatch = new ArrayList<>();
        int count = 0;

        for (Map.Entry<String, List<RawJsonDataDto>> entry : batch) {
            try {
                saveSingleMember(entry.getKey(), entry.getValue());
                count++;

                // Flush and clear every batchSize records to free memory
                // and actually trigger the JDBC batch to the DB
                if (count % batchSize == 0) {
                    entityManager.flush();
                    entityManager.clear();
                }

            } catch (Exception e) {
                log.error("[RxBatchSaverService.saveBatch] Failed for memberId={}: {}",
                        entry.getKey(), e.getMessage());
                failedInBatch.add(entry.getKey());
            }
        }

        // Flush remaining records in the batch
        entityManager.flush();
        entityManager.clear();

        return failedInBatch;
    }

    private void saveSingleMember(String memberId, List<RawJsonDataDto> records) {

        MembershipInfo member = membershipRepository.findById(Long.parseLong(memberId))
                .orElseGet(() -> {
                    MembershipInfo m = new MembershipInfo();
                    m.setMemberId(Long.parseLong(records.getFirst().memberId()));
                    m.setFirstName(records.getFirst().firstname());
                    m.setLastName(records.getFirst().lastname());
                    m.setDob(records.getFirst().dob());
                    return m;
                });

        Set<String> existingRx = member.getRxInfo().stream()
                .map(RxInfo::getRx)
                .collect(Collectors.toSet());

        records.stream()
                .filter(r -> !existingRx.contains(r.rx()))
                .map(r -> buildRxInfo(r, member))
                .forEach(rx -> member.getRxInfo().add(rx));

        // persist/merge into the EntityManager's batch queue
        // actual SQL is deferred until flush()
        entityManager.merge(member);
    }

    private RxInfo buildRxInfo(RawJsonDataDto record, MembershipInfo member) {
        RxInfo rxInfo = new RxInfo();
        rxInfo.setRx(record.rx());
        rxInfo.setDrugName(record.drugName());
        rxInfo.setDescription(record.description());
        rxInfo.setMember(member);
        return rxInfo;
    }
}
```

###### How it all fits together



```
10,000 records parsed
        │
        ▼
grouped by memberId (LinkedHashMap preserves order)
        │
        ▼
partitioned into chunks of batchSize (e.g. 100 members)
        │
   ┌────┴────┐
Batch 1   Batch 2  ...  Batch N       ← each in REQUIRES_NEW transaction
   │
per-member try/catch
   ├── success → entityManager.merge() queued
   └── failure → memberId added to failedList, continues
        │
entityManager.flush() → single JDBC batch roundtrip to DB
entityManager.clear() → frees memory
```



The key insight is `entityManager.merge()` doesn't hit the DB immediately — it queues the SQL. Only when `flush()` is called does Hibernate send the accumulated INSERTs to the DB in one JDBC batch, giving you the network efficiency you need for 100k+ records.

#### Claude responded: Great question.


Great question. Here's the breakdown:

---



###### `saveAll()` — Spring Data JPA convenience method

 java

```java
membershipRepository.saveAll(memberList);
```



Internally, `saveAll()` just loops and calls `save()` (which calls `entityManager.persist()` or `merge()`) on each entity. It **does** respect Hibernate batch settings **if** your config is correct:

 properties

```properties
spring.jpa.properties.hibernate.jdbc.batch_size=100
spring.jpa.properties.hibernate.order_inserts=true
```



But there are silent killers that disable batching under the hood:

- Using `GenerationType.IDENTITY` — Hibernate must fire an immediate INSERT to get the generated ID back, so it **can't batch**
- Mixed entity types in the same flush — Hibernate breaks the batch if types aren't ordered
- No `flush()` + `clear()` between chunks — all entities pile up in the persistence context, causing `OutOfMemoryError` on large files

---



###### `saveSingleMember` + `entityManager.merge()` + manual `flush/clear` — explicit control

 java

```java
entityManager.merge(member);   // queued, no DB hit yet

if (count % batchSize == 0) {
    entityManager.flush();     // NOW sends the accumulated batch to DB
    entityManager.clear();     // frees memory, detaches all entities
}
```



You're manually controlling **when** SQL is sent and **when** memory is freed.

---



###### Side-by-side comparison



|  | `saveAll()` | `merge()` + manual `flush/clear` |
| --- | --- | --- |
| Batching works automatically | Only if config is perfect + SEQUENCE id | Always explicit |
| Memory management | None — all entities stay in persistence context | You control it with `clear()` |
| Safe for 1,000,000 records | No — will OOM | Yes |
| Per-record error isolation | Hard — one exception aborts the loop | Easy — wrap each `merge()` in try/catch |
| Code complexity | Simple | More verbose |
| Requires SEQUENCE id strategy | Yes, for batching to actually work | Yes, same requirement |

---



###### The real gotcha with `saveAll()`

 java

```java
// This looks like a batch insert but may fire 10,000 individual INSERTs
// if IDENTITY generation is used — Hibernate silently disables batching
membershipRepository.saveAll(tenThousandMembers);
```



Hibernate won't warn you. You only find out by enabling SQL logging and counting the statements.

---



###### Bottom line

- **Small datasets, simple code** → `saveAll()` is fine, just fix the ID strategy
- **Large datasets (100k+), fault isolation needed** → `merge()` + manual `flush/clear` is the right tool because you control memory, you control when the batch fires, and you can catch failures per record without aborting the whole operation

private ValidAndInvalidRecordsDto parseAndValidateFileRecords(MultipartFile file) {
    List<IndexedRecord> validList = new ArrayList<>();
    List<String> invalidList = new ArrayList<>();

    int lineNumber = 0;

    try (BufferedReader reader = new BufferedReader(new InputStreamReader(file.getInputStream()))) {
        String line;
        while ((line = reader.readLine()) != null) {
            lineNumber++;
            String trimmedLine = line.trim();
            if (trimmedLine.isEmpty()) continue;

            try {
                RawJsonDataDto record = objectMapper.readValue(trimmedLine, RawJsonDataDto.class);
                if (isValidRecord(record))
                    validList.add(new IndexedRecord(lineNumber, record));
                else
                    invalidList.add("Line " + lineNumber + ": " + record);

            } catch (JacksonException e) {

    return new ValidAndInvalidRecordsDto(validList, invalidList);
}

            .collect(Collectors.groupingBy(

                    Collectors.toList()

                    .collect(Collectors.toSet());

            List<RxInfo> newRxList = indexedRecords.stream()
                    .filter(indexed -> !existingRxInfo.contains(indexed.data().rx()))
                    .map(indexed -> getRxInfo(indexed.data(), member))
                    .toList();

            member.getRxInfo().addAll(newRxList);
            membershipRepository.save(member);

        } catch (Exception e) {
            // Report every line number that belonged to this failed memberId
            indexedRecords.forEach(indexed ->
                failedRecords.add("Line " + indexed.lineNumber() + " (memberId=" + memberId + "): " + e.getMessage())
            );
            log.error("[RxServiceImpl.saveToDatabase] Failed to save memberId={}: {}", memberId, e.getMessage());
        }
    });

    return failedRecords;

@Service
@Slf4j
public class RxBatchSaverService {

    private final MembershipInfoRepository membershipRepository;
    private final EntityManager entityManager;

    @Value("${rx.batch.size:100}")
    private int batchSize;

    public RxBatchSaverService(MembershipInfoRepository membershipRepository,
                               EntityManager entityManager) {
        this.membershipRepository = membershipRepository;
        this.entityManager = entityManager;
    }

    public List<String> batchInsert(List<IndexedRecord> records) {

        Map<String, List<IndexedRecord>> groupedByMemberId = records.stream()
                .collect(Collectors.groupingBy(
                        indexed -> indexed.data().memberId(),
                        LinkedHashMap::new,
                        Collectors.toList()
                ));

        List<Map.Entry<String, List<IndexedRecord>>> entries =
                new ArrayList<>(groupedByMemberId.entrySet());

        List<String> failedRecords = new ArrayList<>();

        for (int i = 0; i < entries.size(); i += batchSize) {
            List<Map.Entry<String, List<IndexedRecord>>> batch =
                    entries.subList(i, Math.min(i + batchSize, entries.size()));

            log.info("[RxBatchSaverService.batchInsert] Inserting batch {}-{} of {} members",
                    i + 1, i + batch.size(), entries.size());

            List<String> batchFailures = saveBatch(batch);
            failedRecords.addAll(batchFailures);
        }

        return failedRecords;
    }

    @Transactional(Transactional.TxType.REQUIRES_NEW)
    public List<String> saveBatch(List<Map.Entry<String, List<IndexedRecord>>> batch) {

        List<String> failedInBatch = new ArrayList<>();
        int count = 0;

        for (Map.Entry<String, List<IndexedRecord>> entry : batch) {
            String memberId = entry.getKey();
            List<IndexedRecord> indexedRecords = entry.getValue();

            try {
                saveSingleMember(memberId, indexedRecords);
                count++;

                if (count % batchSize == 0) {
                    entityManager.flush();
                    entityManager.clear();
                }

            } catch (Exception e) {
                // Report every line number that belonged to this failed memberId
                indexedRecords.forEach(indexed ->
                    failedInBatch.add("Line " + indexed.lineNumber() + " (memberId=" + memberId + "): " + e.getMessage())
                );
                log.error("[RxBatchSaverService.saveBatch] Failed for memberId={}: {}", memberId, e.getMessage());
            }
        }

        entityManager.flush();
        entityManager.clear();

        return failedInBatch;
    }

    private void saveSingleMember(String memberId, List<IndexedRecord> indexedRecords) {

        MembershipInfo member = membershipRepository.findById(Long.parseLong(memberId))
                .orElseGet(() -> {
                    RawJsonDataDto first = indexedRecords.getFirst().data();
                    MembershipInfo m = new MembershipInfo();
                    m.setMemberId(Long.parseLong(first.memberId()));
                    m.setFirstName(first.firstname());
                    m.setLastName(first.lastname());
                    m.setDob(first.dob());
                    return m;
                });

        Set<String> existingRx = member.getRxInfo().stream()
                .map(RxInfo::getRx)
                .collect(Collectors.toSet());

        indexedRecords.stream()
                .filter(indexed -> !existingRx.contains(indexed.data().rx()))
                .map(indexed -> buildRxInfo(indexed.data(), member))
                .forEach(rx -> member.getRxInfo().add(rx));

        entityManager.merge(member);
    }

    private RxInfo buildRxInfo(RawJsonDataDto record, MembershipInfo member) {
        RxInfo rxInfo = new RxInfo();
        rxInfo.setRx(record.rx());
        rxInfo.setDrugName(record.drugName());
        rxInfo.setDescription(record.description());
        rxInfo.setMember(member);
        return rxInfo;
    }
}
